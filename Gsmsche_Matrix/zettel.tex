		$v_1,\dots,v_n \in  (\quad, \langle .\rangle)$
		\[
			G (v_1,\dots,v_n) = ( \langle v_i,v_j\rangle)_{i,j\leq k}
		.\]

	\end{definition}

	\[
		\det G (v_1,\dots,v_n) = \Vol_k(P(v_1,\dots,v_n))^2
