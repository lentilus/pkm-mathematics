	.\]

	\begin{definition}[Duale Basis]
		\[
			\varphi^{\mathcal B}_{e_1},\dots,\varphi^{\mathcal B}_{e_n} \tx{Duale Basis zu $\mathcal B$ }
		.\]
	\end{definition}
