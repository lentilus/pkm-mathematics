\begin{zettel}{Lipschitz-stetig}
\begin{flashcard}
    \begin{definition}[Lipschitz-stetig]
        $f$ ist Lipschitz-stetig, falls
        \[
            \exists L > 0 d_Y (f(x_1),f(x_2)) \leq L d_X (x_1,x_2) 
        .\]
        $L$ - Lipschitz-Konstante
    \end{definition}
\end{flashcard}
