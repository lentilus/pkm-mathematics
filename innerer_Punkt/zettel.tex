	\begin{definition}
		Sei $Y \subset X$ $(X, \tau )$ - top. Raum $x \in  Y$ heisst innerer Punkt, falls eine Umgebung $U$ von $x$ existiert sd $U \subset Y$
		\[
			\dot{Y} = \set{y \in  Y}{y \tx{ - innerer Punkt } }
		.\]
	\end{definition}
