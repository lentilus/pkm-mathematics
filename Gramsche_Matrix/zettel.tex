\begin{zettel}{Gramsche Matrix}
\begin{flashcard}
    \begin{definition}[Gramsche Matrix]
\[
    G( v_1,\dots,v_k ):= \left(\langle v_i,v_j\rangle \right)_{i,j = 1,\dots,k}
\] ist die Gramsche Matrix.
    \end{definition}
\end{flashcard}

\begin{theorem}
\[
\det(G ( v_1,\dots,v_k)) = \Vol_k(P(v_1,\dots,v_k))
.\]
\end{theorem}
