	\begin{definition}[Norm]
		\begin{enumerate}
			\item $ ||x|| = 0 \iff x = 0$
			\item $ ||\lambda x|| = |\lambda| ||x||$
			\item $ ||x + y|| \leq  ||x|| + ||y||$
		\end{enumerate}
	\end{definition}
\end{flashcard}
