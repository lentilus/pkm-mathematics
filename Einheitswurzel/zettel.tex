\begin{zettel}{Einheitswurzel}
\begin{flashcard}
    \begin{definition}[Einheitswurzel]
        Wir nennen $x$ eine Einheitswurzel, wenn
        \[
            \exists r : x^r = 1
        .\]
    \end{definition}
    \begin{example}[Einheitswurzeln]
        Elemente auf dem Einheitskreis
        \begin{enumerate}
            \item $i$ 
            \item $-i$ 
            \item $-1$ 
            \item $1$ 
        \end{enumerate}
    \end{example}

\end{flashcard}
