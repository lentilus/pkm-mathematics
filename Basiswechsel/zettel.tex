\begin{zettel}{Basiswechsel}
    Sei $E = (e_1 , e_2, e_3 )$ und $v,w$ lin. unabh"angig.
    \[
         B = (v,w, v \times w) \tx{ ist eine Basis} 
    .\]

    \[
        T^B_{E} = (v|w|v \times w)
    .\]

    \[
        \det(T^B_E ) = \det((T^{B}_E )^t ) \stackrel{(a)}{=}  \langle v \times w, v \times w\rangle  > 0
    .\]
\end{zettel}
