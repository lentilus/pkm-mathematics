\begin{flashcard}
    \begin{definition}[Untergruppe]
        $H \subset G$ ist eine Untergruppe falls
    \begin{enumerate}
        \item $H \neq \emptyset$ 
        \item $\forall a,b \in  H \colon a \cdot b \in  H$ 
        \item $\forall a \in  H \colon a^{-1} \in  H$ 
    \end{enumerate}
\end{definition}

\end{flashcard}
