	\begin{definition}[Abbildung]
		\[
			f \from K^n\to  K^n \tx{mit} f \begin{pmatrix}
				x_1    \\
				\vdots \\
				x_n    \\
			\end{pmatrix} =
			\begin{pmatrix}
				f_1 (x_1, \dots , x_n) \\
				\vdots                 \\
				f_m (x_1, \dots, x_n)  \\
			\end{pmatrix} \tx{ ist eine Abbildung }
		.\]

	\end{definition}
\end{flashcard}
