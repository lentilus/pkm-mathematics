	\end{question}
	\begin{definition}[Skalarprodukt komplexe VR]
		Eine Abbildung
		\[
			\langle .,.\rangle : \C \times \C \to \C
		.\]
		ist ein Sklarprodukt falls sie sesquilinear, hermitesch und positiv definit ist.
	\end{definition}
\end{flashcard}
