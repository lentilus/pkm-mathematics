\begin{flashcard}
    \begin{definition}[erzeugte Gruppe]
        Sei $T \subset G$, dann ist die von $T$ erzeugte Gruppe $ \langle T\rangle $ die kleinste Gruppe die $T$ enth"alt.
        Bemerke, dass
\[
        \langle T\rangle  = \bigcap \set{H \leq G}{T \subset H}
.\]
\end{definition}

\end{flashcard}
